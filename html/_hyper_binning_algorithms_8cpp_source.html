293  int dim = binningRange.getDimension();

295  HyperBinningMaker* binnningMaker = 0;
296 
297  int startDimension = 0;
298 
299  if (optExist(AlgOption::START_DIM ))
300  startDimension = getOpt(AlgOption::START_DIM ).getIntOpt();
301 
302  int numPhaseBinPairs = 3;
303  if (optExist(AlgOption::NUM_BIN_PAIRS ))
304  numPhaseBinPairs = getOpt(AlgOption::NUM_BIN_PAIRS ).getIntOpt();
305 
306 
307 
308  //Chose the binning algorithm
309 
310  if (_alg == SMART ) {
311  binnningMaker = new HyperBinningMakerSmart(binningRange, points, startDimension);
312  }
313  if (_alg == MINT ) {
314  binnningMaker = new HyperBinningMakerMint(binningRange, points, startDimension);
315  }
316  if (_alg == MINT_SMART ) {
317  binnningMaker = new HyperBinningMakerMintSmart(binningRange, points, startDimension);
318  }
319  if (_alg == MINT_RANDOM ) {
320  binnningMaker = new HyperBinningMakerMintRandomise(binningRange, points);
321  }
322  if (_alg == SMART_RANDOM ) {
323  binnningMaker = new HyperBinningMakerSmartRandomise(binningRange, points);
324  }
325  if (_alg == LIKELIHOOD ) {
326  binnningMaker = new HyperBinningMakerLikelihood(binningRange, points);
327  }
328  if (_alg == SMART_LIKELIHOOD ) {
329  binnningMaker = new HyperBinningMakerSmartLikelihood(binningRange, points);
330  }
331  if (_alg == SMART_MULTI) {
332  binnningMaker = new HyperBinningMakerMultiSmart(binningRange, points, startDimension);
333  }
334  if (_alg == FUNC_PHASE) {
335  HyperBinningMakerPhaseBinning* phaseBinningMaker = new HyperBinningMakerPhaseBinning(binningRange, 0);
336  phaseBinningMaker->setNumBinPairs(numPhaseBinPairs);

339  phaseBinningMaker->setBinEdges( getOpt(AlgOption::PHASE_BIN_EDGES ).getDoubleVectorOpt() );
340

342  }
343  //Now set the options
344 
345 
346  if (optExist(AlgOption::RAND_SEED ))

351 
352  if (optExist(AlgOption::BINNING_DIMS ))
353  binnningMaker->setBinningDimensions( getOpt(AlgOption::BINNING_DIMS).getIntVectorOpt() );
354 
355  if (optExist(AlgOption::FUNC ))
356  binnningMaker->setHyperFunction( getOpt(AlgOption::FUNC).getFuncOpt() );
357 
358  if (optExist(AlgOption::MIN_BIN_WIDTH )){
359  //can provide either a double or a HyperPoint here.
360 
361  HyperPoint widthsA = getOpt(AlgOption::MIN_BIN_WIDTH).getHyperPointOpt();
362  double widthsB = getOpt(AlgOption::MIN_BIN_WIDTH).getDoubleOpt ();
363  if ( widthsA.getDimension() == 0 ) widthsA = HyperPoint(dim, widthsB);
364  binnningMaker->setMinimumEdgeLength(widthsA);
365  }
366 
367  if (optExist(AlgOption::USE_SHADOW_DATA ))
368  binnningMaker->addShadowHyperPointSet( getOpt(AlgOption::USE_SHADOW_DATA).getHyperPointSetOpt() );
369 
370  if (optExist(AlgOption::MIN_BIN_CONTENT ))
371  binnningMaker->setMinimumBinContent( getOpt(AlgOption::MIN_BIN_CONTENT).getDoubleOpt() );
372 
373  if (optExist(AlgOption::MIN_SHADOW_BIN_CONTENT ))
374  binnningMaker->setShadowMinimumBinContent( getOpt(AlgOption::MIN_BIN_CONTENT).getDoubleOpt() );
375 
376  if (optExist(AlgOption::DRAW_ALGORITHM ))

378 
379  if (optExist(AlgOption::AXIS_NAMES ))
380  binnningMaker->setNames( getOpt(AlgOption::AXIS_NAMES).getHyperNameOpt() );
381 
382  if (optExist(AlgOption::SNAP_TO_GRID )){
383  binnningMaker->useSnapToGrid( getOpt(AlgOption::SNAP_TO_GRID).getBoolOpt() );
384  }
385 
386  if (optExist(AlgOption::GRID_MULTIPLIER )){
387 
388  HyperPoint valA = getOpt(AlgOption::GRID_MULTIPLIER).getHyperPointOpt();
389  int valB = getOpt(AlgOption::GRID_MULTIPLIER).getIntOpt ();
390

392  binnningMaker->setGridMultiplier(valA);
393  }

395  binnningMaker->setGridMultiplier(valB);
396  }
