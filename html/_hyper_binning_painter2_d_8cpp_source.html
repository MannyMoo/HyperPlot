89 int HyperBinningPainter2D::getFillColour(double binContents){
90  double min = _histogram->getMin();
91  double max = _histogram->getMax();
92  if (_density == true){
93  min = _histogram->getMinDensity();
94  max = _histogram->getMaxDensity();
95  }
96 
97  if (binContents < min || binContents > max) return kWhite;
98 
99  int ndivz = gStyle->GetNumberContours();
100  double dz = max - min;
101  double scale = ndivz/dz;
102  int ncolors = gStyle->GetNumberOfColors();
103 
104  int color = int(0.01+(binContents-min)*scale);
105  int theColor = int((color+0.99)*double(ncolors)/double(ndivz));
106  if (theColor > ncolors-1) theColor = ncolors-1;
107 
108  return gStyle->GetColorPalette(theColor);
